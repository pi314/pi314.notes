語法
-----------------------------------------------------------------------------

Assign value to an variable ::

  let var: u32 = 42;
  let mut str = String::new();

loop ::

  loop {
      break;
  }

match ::

  match val {
      A(a) => value_a,
      B(b) => value_b,
      C(c) => {
          expr;
          expr;
      }
  }

::

  foo();                      // call a function
  println!("Hello world!");   // call a macro
  print!("> ");               // print without newline
